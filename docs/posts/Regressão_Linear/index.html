Nesta atividade aplico a regressão linear para analisar dados e identificar a tendência entre as variáveis fornecidas. A técnica permite criar um modelo simples capaz de prever valores futuros a partir do comportamento observado nos dados.

import numpy as np
import plotly.graph_objects as go

# Carregar os dados a partir dos arquivos .txt
x = np.loadtxt('x.txt')
y = np.loadtxt('y.txt')

# Regressão linear (forma matricial)
X = np.vstack([np.ones(len(x)), x]).T  # Matriz de design com intercepto
beta = np.linalg.inv(X.T @ X) @ X.T @ y  # Coeficientes (intercepto e inclinação)
intercept, slope = beta

# Gerar valores previstos
y_pred = intercept + slope * x

# Criar o gráfico
fig = go.Figure()
fig.add_trace(go.Scatter(x=x, y=y, mode='markers', name='Dados'))
fig.add_trace(go.Scatter(x=x, y=y_pred, mode='lines', name='Regressão Linear'))

fig.update_layout(
    title='Regressão Linear de x e y',
    xaxis_title='x',
    yaxis_title='y',
    template='plotly_white'
)

fig.show()
